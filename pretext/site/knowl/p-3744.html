\begin{align*} \Set{(x,y,z)}{\theta(x,y,z)=0} &=\text{the half of the $xz$-plane with } x \gt 0\\ \Set{(x,y,z)}{\theta(x,y,z)=\frac{\pi}{4}} &=\text{the half of the plane $y=x$ with } x \gt 0 \end{align*}
