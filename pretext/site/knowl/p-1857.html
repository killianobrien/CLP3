\begin{equation*} f\big(1+\dee{x}\,,\,\dee{y}\big) \approx f(1,0) + \frac{\dee{x}}{e^{f(1,0)}+0} + \frac{1-f(1,0)}{e^{f(1,0)}+0} \,\dee{y} = \dee{x} +\dee{y} \end{equation*}

\begin{equation*} f\big(0.99\,,\,0.01\big) \approx -0.01 + 0.01 =0 \end{equation*}
