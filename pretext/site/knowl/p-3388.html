\begin{align*} \int_0^2\int_{-\sqrt{2x-x^2}}^{\sqrt{2x-x^2}} \big(2x+3y\big)\dee{y}\,\dee{x} &=2\dblInt_\cR x\ \dee{x}\,\dee{y} +3\dblInt_\cR y\ \dee{x}\,\dee{y}\\ &=2\,\bar x\, \text{Area}(\cR) +3\,\bar y\, \text{Area}(\cR) =2\pi \end{align*}
